(defn write-bag!
  [bag-writer bag]
  {:pre [(satisfies? bag-io/BagWriter bag-writer)
         (bag.check/bag? bag)]}
  (bag-io/write-bag bag-writer bag))

						
			

			

				
					
Write a BAG to a CAR file.

						
				
					
(defn write-to-file!
  [bag file-name]
  {:pre [(string? file-name)]}
  (let [car-writer (car.file/->CarFile file-name)]
    (bag-io/write-bag car-writer bag)))
