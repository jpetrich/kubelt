(deftest from-json-test
  (testing "parse empty object"
    (let [json-str "{}"
          keywordize? false
          data (json/from-json json-str keywordize?)]
      (is (= {} data)
          "parsed JSON string object becomes map")))
  (testing "parse empty array"
    (let [json-str "[]"
          keywordize? false
          data (json/from-json json-str keywordize?)]
      (is (= [] data)
          "parsed JSON string array becomes vector"))))
