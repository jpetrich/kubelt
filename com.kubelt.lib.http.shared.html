Return the request method from a request map in the format expected by the Node.js http package.

 
(defn request->method
  [m]
  {:pre [(map? m)]}
  (cstr/upper-case (name (:http/method m))))
 
(defn request->domain
  [m]
  {:pre [(map? m)]}
  (:uri/domain m))
 
(defn request->port
  [m]
  {:pre [(map? m)]}
  (:uri/port m))
 
(defn request->path
  [m]
  {:pre [(map? m)]}
  (:uri/path m))
 
(defn request->headers
  [m]
  {:pre [(map? m)]}
  (let [headers (get m :http/headers {})]
    headers))
 
(defn request->body
  [m]
  {:pre [(map? m)]}
  (:http/body m))
 
(defn request->scheme
  [m]
  {:pre [(map? m)]}
  (name (get m :uri/scheme )))
