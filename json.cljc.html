Recursively transform string map keys using the given transform fn.

						
				
					
(defn- transform-keys
  [m transform-fn]
  (let [f (fn [[k v]]
            (if (string? k)
              [(transform-fn k) v]
              [k v]))]
    (walk/postwalk (fn [x]
                     (if (map? x)
                       (into {} (map f x))
                       x)) m)))

Parse a JSON string into edn data.

						
				
					
(defn json-str->edn
  ([s]
   #?(:clj
      (json/read-value s)
      :cljs
      (let [r (transit/reader :json)]
        (transit/read r s))))
  ([s mapper]
   #?(:clj
      (json/read-value s keyword-mapper)
      :cljs
      (let [r (transit/reader :json)
            decode-fn (get mapper :decode-key-fn)
            data-edn (transit/read r s)]
        (transform-keys data-edn decode-fn)))))

						
			

			

				
					
Parse a JSON string into edn data. If the keywordize? flag is set, process string keys into keywords in the result.

						
				
					
(defn from-json
  [body keywordize?]
  (if keywordize?
    (json-str->edn body keyword-mapper)
    (json-str->edn body)))
