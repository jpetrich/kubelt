(defn- get-storage
  []
  (.-localStorage js/window))

Set key' in browser's localStorage to val`.

 
(defn- set-item!
  [key val]
  (.setItem (get-storage) key val))

Returns value of `key' from browser's localStorage.

 
(defn- get-item
  [key]
  (.getItem (get-storage) key))

Remove the browser's localStorage value for the given key

 
(defn- remove-item!
  [key]
  (.removeItem (get-storage) key))
 
(def ^:private DB_ID_KEY "com.kubelt.lib.storage.browser.db")

Return a function that stores SDK state (supplied as a map) into a browser local storage location/key. This function returns a promise that resolves to map describing what was written:

  • :data, the map of data that was written
 
(defn- store-fn
  [m]
  {:pre [(map? m)] :post [(lib.promise/promise? %)]}
  (lib.promise/promise
   (fn [resolve _]
     (let [data (lib.json/edn->json-str m)]
       (set-item! DB_ID_KEY data)
       (resolve {:data data})))))

Return a function that loads SDK state from a browser local storage location/key and returns it as a map.

 
(defn- restore-fn
  []
  {:post [(lib.promise/promise? %)]}
  (lib.promise/promise
   (fn [resolve _]
     (let [data (-> (get-item DB_ID_KEY)
                    (lib.json/json-str->edn))]
       (resolve data)))))

Create a configuration storage capability.

 
(defn create
  [app-name]
  (tap> [::create app-name])
  {:com.kubelt/type :kubelt.type/storage
   :storage/store-fn store-fn
   :storage/restore-fn restore-fn})
