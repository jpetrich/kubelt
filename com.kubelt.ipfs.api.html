(defn paths
  [m]
  (letfn [(f [x]
            (cond
              (keyword? x) x
              (vector? x) x
              (map? x)
              (if (= :kubelt.type/api-resource (get x :com.kubelt/type))
                nil
                x)))]
    (walk/postwalk f m)))
