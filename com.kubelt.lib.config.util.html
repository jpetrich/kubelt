Convert a log level string (e.g. 'info', ':info') to a keyword.

						
				
					
(defn str->kw
  [log-level]
  (if (string? log-level)
    (keyword (-> log-level (cstr/trim) (cstr/replace #"^:" "")))
    log-level))
