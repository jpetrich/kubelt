(deftest method-test
  (testing "get method from request map"
    (let [request {:http/method :get}
          result (lib.shared/request->method request)]
      (is (string? result)
          "method is a string")
      (is (= "GET" result)
          "method has correct name"))))

  (testing "basic http URL"
    (let [request {:uri/scheme :http
