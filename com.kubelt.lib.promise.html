(defn- cast-as-array
  [coll]
  (if (or (array? coll)
          (not (reduceable? coll)))
    coll
    (into-array coll)))

Returns true if the argument is a JavaScript promise, and false otherwise.

						
				
					
(defn promise?
  [x]
  (= js/Promise (type x)))

						
			

			

				
					
Return a promise that uses the given resolver function. This function must accept two arguments, a function that may be called to resolve the promise, and a second function that may be called to reject the promise.

						
				
					
(defn promise
  [resolver]
  {:pre [(fn? resolver)]}
  (js/Promise. resolver))

						
			

			

				
					
				
				
					
(defn then
  ([promise on-fulfilled]
   {:pre [(fn? on-fulfilled)]}
   (.then promise on-fulfilled))
  ([promise on-fulfilled on-rejected]
   {:pre [(every? fn? [on-fulfilled on-rejected])]}
   (.then promise on-fulfilled on-rejected)))

						
			

			

				
					
				
				
					
(defn catch
  [promise on-rejected]
  {:pre [(fn? on-rejected)]}
  (.catch promise on-rejected))

						
			

			

				
					
				
				
					
(defn finally
  [promise on-final]
  {:pre [(fn? on-final)]}
  (.finally promise on-final))

						
			

			

				
					
Return a promise that resolves to the given value.

						
				
					
(defn resolved
  [x]
  (.resolve js/Promise x))

						
			

			

				
					
Return a promise that rejects to the given value.

						
				
					
(defn rejected
  [x]
  (.reject js/Promise x))

						
			

			

				
					
Returns a promise that resolves after the given millisecond interval.

						
				
					
(defn timeout
  [ms]
  (promise (fn [resolve _]
             (js/setTimeout resolve ms))))

						
			

			

				
					
Takes an iterable of promises and returns a single promise that resolves as soon as any of the promises in the iterable fulfills, with the value of the fulfilled promise.
https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Promise/any

						
				
					
(defn any
  [coll]
  (.any js/Promise (cast-as-array coll)))

Resolves when all promise values in the map resolve. The resolved value will e a map with the same keys as the input map, but with the resolved values of the promises. Rejects with first error if any promise rejects. Like Promise.all() but for Maps.
Google Clojure library promise utilities:   https://google.github.io/closure-library/api/goog.async.promises.html JavaScript native Maps:   https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Map

						
				
					
(defn all-map
  [js-map]
  {:pre [(= js/Map (type js-map))] :post [(promise? %)]}
  (allMapValues js-map))
