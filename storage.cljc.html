storage?

 

						
			

			

				
					
Returns true if the given value is a map with a storage type tag, and false otherwise.

						
				
					
(defn storage?
  [x]
  (and
   (map? x)
   (= :kubelt.type/storage (get x :com.kubelt/type))))

Return true if the given value conforms to the schema for a storage capabililty map, and false otherwise.

						
				
					
(defn valid?
  [x]
  (malli/validate spec.storage/storage x))

Create a platform-appropriate configuration storage. This initializes the storage and returns a wrapper that may be injected into the SDK init fn as a dependency to allow for SDK state to be stored and retrieved.

						
				
					
(defn create
  [app-name]
  {:pre [(string? app-name)]}
  #?(:browser (storage.browser/create app-name)
     :jvm (storage.jvm/create app-name)
     :node (storage.node/create app-name)))
